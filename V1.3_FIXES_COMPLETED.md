**File Updated**: `components/path-builder.tsx`
**Implementation**:

```tsx

  let value = parseFloat(e.target.value) || 0
  // Cap the value at 30% max
  if (value > 30) {
    value = 30
    e.target.value = '30'
  }
  if (value < 0) {
    value = 0
    e.target.value = '0'
  }
  // ... rest of handler
}}
```
